model
{
    for (i in 1:I) {
        m[i] ~ dnorm(theta, tau.I)
        for (j in 1:J) {
            y[i, j] ~ dnorm(m[i], tau.J)
            for (k in 1:K) {
                z[i, j, k] ~ dnorm(y[i, j], tau.K)
            }
        }
    }
    sigma2.I <- 1/tau.I
    tau.I ~ dgamma(0.001, 0.001)
    sigma2.J <- 1/tau.J
    tau.J ~ dgamma(0.001, 0.001)
    sigma2.K <- 1/tau.K
    tau.K ~ dgamma(0.001, 0.001)
    theta ~ dflat()
}
